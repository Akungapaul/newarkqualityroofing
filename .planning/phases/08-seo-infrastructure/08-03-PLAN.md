2. **Anchor text rotation utility** -- create `src/data/linking/anchor-text.ts`:
   ```typescript
   const servicePatterns = [
     (name: string) => `${name.toLowerCase()} services`,
     (name: string) => `professional ${name.toLowerCase()}`,
     (name: string) => `${name.toLowerCase()} in Essex County`,
     (name: string) => `expert ${name.toLowerCase()}`,
   ];

   const cityPatterns = [
     (name: string) => `${name} roofing`,
     (name: string) => `roofing services in ${name}`,
     (name: string) => `${name}, NJ roofing`,
     (name: string) => `roofing contractor in ${name}`,
   ];

   export function getServiceAnchorText(name: string, index: number): string {
     return servicePatterns[index % servicePatterns.length](name);
   }

   export function getCityAnchorText(name: string, index: number): string {
     return cityPatterns[index % cityPatterns.length](name);
   }
   ```
